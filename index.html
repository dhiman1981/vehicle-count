      function n(id){
      const v = parseInt(document.getElementById(id).value, 10);
      return Number.isFinite(v) && v > 0 ? v : 0;
      }
      function generateMessage() {
      const counts = {
      Car: n('car'),
      Activa: n('activa'),
      Motorbike: n('motorbike'),
      'Tractor Trolley': n('tractor'),
      Bus: n('bus'),
      Cruiser: n('cruiser'),
      'Helpline Car': n('helpline'),
      Cycle: n('cycle'),
      Auto: n('auto')
      };
      // keep only non-zero lines
      const nonZero = Object.entries(counts).filter(([_, c]) => c > 0);
      if (nonZero.length === 0) {
      alert('Please enter at least one count > 0.');
      return;
      }
      // India day name + date + time
      const now = new Date();
      const weekday = new Intl.DateTimeFormat('en-IN', { weekday: 'long', timeZone: 'Asia/Kolkata' }).format(now);
      const datePart = new Intl.DateTimeFormat('en-IN', { day: '2-digit', month: 'short', year: 'numeric', timeZone: 'Asia/Kolkata' }).format(now);
      const timePart = new Intl.DateTimeFormat('en-IN', { hour: '2-digit', minute: '2-digit', hour12: true, timeZone: 'Asia/Kolkata' }).format(now);
      const stamp = `${weekday}, ${datePart} ${timePart}`;
      let total = 0;
      let message = `Radha Soami Ji 🙏\n\n🚗 Vehicle Count Report\n📅 ${stamp}\n\n`;
      nonZero.forEach(([k, v]) => { message += `${k}: ${v}\n`; total += v; });
      message += `\n✅ Total: ${total}`;
      // show preview on page so you can confirm changes
      document.getElementById('preview').textContent = message;
      // open WhatsApp (mobile friendly)
      const encoded = encodeURIComponent(message);
      window.location.href = `https://api.whatsapp.com/send?text=${encoded}`;
      }
      </script>
